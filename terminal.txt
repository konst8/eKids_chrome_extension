// show current path

pwd

// go one level up (to parent directory)

cd ..

// go to path

cd <path>
for example 
cd eKids_chrome_extension

// list files in current folder

ls


////// git

// init version control for current folder

git init

// save initial version of your code

git add --all && git commit -a -m "initial commit"

// show current status of your code

git status

// show changes (difference) in our code

git diff

// show histrory of changes (commits)

git log

// clone (download) git repository (folder) from github

git clone <path_to_repository>
for example:
git clone https://github.com/konst8/eKids_chrome_extension.git
